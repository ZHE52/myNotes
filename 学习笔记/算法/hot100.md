### 两数之和
``` javascript
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
  const id=new Map()// 创建一个空哈希表
  for(let j=0;;j++){// 枚举 j
         const x=nums[j]      // 在左边找 nums[i]，满足 nums[i]+x=target
        if(id.has(target-x)){// 找到了
            return [id.get(target-x),j] // 返回两个数的下标
        }  
    id.set(x,j)  // 保存 nums[j] 和 j
  }
   
};
```

这里为啥会想到用map，就是因为数组的原因吗，要考虑下标和值都要存，所以用map
